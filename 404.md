document.onreadystatechange = function () {
  if (document.readyState == 'complete') {
    var uriPath = decodeURIComponent(location.href).split('/');
    var postId = uriPath[uriPath.length - 2];
    if (postId.match(/^[0-9][0-9][0-9]$/)) {
      $('#alpha-msg').html('<p>Sorry, the post you are looking for (<a href="' + location.href +
      '">' + location.href + '</a>) is outdated and has been removed from this site.</p>' +
      '<p>Thank you for your interest so far.</p>');
    }
  }
};
